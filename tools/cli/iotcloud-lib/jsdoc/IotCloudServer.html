Class: IotCloudServer

IotCloudServer

new IotCloudServer(target, config)

Parameters:
Name Type Description
target OcfDevice | object Target device information.
Properties
Name Type Description
deviceId string Device ID if target's type is 'object'
nick string Device nick name if target's type is 'object'
ocfDeviceType object OCF Device Type if target's type is 'object'
ocfDeviceProfile object The profile of the ocf device.(Will be deprecated)
resources OcfDeivceResource The resource object list of the device.
config object The configuration to communicate Iot Cloud and Auth Provider. The configuration should contains required properties.
Properties
Name Type Description
userId string The id of Samsung Account User(saguid). 10 letters.
accessToken string The access token to use Iot Cloud API.
authCode string The authenticate code for sign-up API.
ssoToken string The sso token to get authCode.
iotCloud object The configuration to communicate with IoT Cloud.
Properties
Name Type Description
uri string The uri of the IoT Cloud.
redirecturi string The redirect uri of the IoT Cloud. It is needed after sign-up.
rejectUnauthorized boolean The flag for ensuring reject during certificate process.
serverCert string The certificate from the IoT Cloud. It is needs to communicate with redirect uri. PEM format.
useRedirection boolean The flag for using redirecturi.
clientKey string The Client private key for mutual certification. PEM format.
clientCert string The Client certificate for mutual certification. PEM format.
authProvider object The configuration to communicate with Auth Provider.
Properties
Name Type Description
uri string The uri of the Auth Provider.
rejectUnauthorized boolean The flag for ensuring reject during certificate process.
iotConsole object The configuration to communicate with Iot Console.
Properties
Name Type Description
uri string The uri of the Iot Console.
rejectUnauthorized boolean The flag for ensuring reject during certificate process.
Example
const vorpal = require('vorpal')();
const IotCloudServer = require('../src/controllers/IotCloudServer');
const CoapMessage = require('../src/models/CoapMessage');
const CoapOptions = require('../src/models/CoapOptions');
const CertManager = require('../src/controllers/CertManager');
const DeviceProfileManager = require('../src/controllers/DeviceProfileManager');
const Configurator = require('../src/controllers/Configurator');
const fs = require('fs');

const config = Configurator.getDefault().getConfig();
const dpm = new DeviceProfileManager(config);
const deviceId = '816be3ba-b993-4ee1-b399-f08d9f001142';
const deviceMnmn = config.mnid;

let power = 'off';

dpm.retrieveDeviceProfile('8e53e69e-91eb-11e7-accc-a3d27f6d0d06')
.then(deviceProfile =>
  DeviceProfileManager.generateOcfDeviceTypeFromDeviceProfile(deviceProfile, deviceMnmn))
.then((ocfDeviceType) => {
  const iotCloudConfig = Configurator.getServerConfig(config, 'iotCloud');
  const cm = new CertManager(config);
  cm.init();
  const { key, cert } = cm.getDeviceKeyCert();
  iotCloudConfig.clientKey = key;
  iotCloudConfig.clientCert = cert;
  const serverConfig = {
    iotCloud: iotCloudConfig,
    authProvider: Configurator.getServerConfig(config, 'authProvider'),
    ssoToken: config.ssoToken,
  };
  const target = {
    deviceId,
    ocfDeviceType,
  };
  const server = new IotCloudServer(target, serverConfig);
  server.get('/capability/switch/main/0', req => (
    new CoapMessage({
      code: CoapMessage.STATUS_CODE.CONTENT,
      token: req.token,
      payload: {
        power,
      },
      options: new CoapOptions({ format: CoapOptions.CONTENT_FORMAT.CBOR }),
    })));
  server.subscribe('/capability/switch/main/0', req => (
    new CoapMessage({
      code: CoapMessage.STATUS_CODE.CONTENT,
      token: req.token,
      payload: {
        power,
      },
      options: new CoapOptions({
        observe: CoapOptions.numberTo4Bytes(0),
        format: CoapOptions.CONTENT_FORMAT.CBOR,
      }),
    })));
  server.post('/capability/switch/main/0', (req) => {
    power = req.payload.power;
    console.log(power);
    setTimeout(() => { server.notify('/capability/switch/main/0', { power }); }, 1000);
    return new CoapMessage({
      code: CoapMessage.STATUS_CODE.CHANGED,
      token: req.token,
      payload: {
        power,
      },
      options: new CoapOptions({ format: CoapOptions.CONTENT_FORMAT.CBOR }),
    });
  });
  return server.start().then(() => server);
})
.then((server) => {
  vorpal
    .command('on', 'switch on')
    .action(function on(args, callback) {
      power = 'on';
      server.notify('/capability/switch/main/0', { power });
      this.log('on');
      callback();
    });

  vorpal
    .command('off', 'switch off')
    .action(function off(args, callback) {
      power = 'off';
      server.notify('/capability/switch/main/0', { power });
      this.log('off');
      callback();
    });

  vorpal
    .delimiter('switch>')
    .show();
});
