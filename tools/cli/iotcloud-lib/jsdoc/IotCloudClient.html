Example
const IotCloudClient = require('./src/controllers/IotCloudClient');
const Configurator = require('./src/controllers/Configurator');
const { getServerConfig } = require('./src/common/util');
const CertManager = require('./src/controllers/CertManager');
const fs = require('./src/controllers/FileSystemHandler');

const config = Configurator.getDefault().getConfig();
const iotCloudConfig = getServerConfig(config, 'iotCloud');
const cm = new CertManager(config);
cm.init();
const { key, cert } = cm.getDeviceKeyCert();
iotCloudConfig.clientKey = key;
iotCloudConfig.clientCert = cert;

const clientConfig = {
  iotCloud: iotCloudConfig,
  ssoToken: config.ssoToken,
  authCode: config.authCode,
  accessToken: config.accessToken,
  userId: config.userId,
  deviceId: config.deviceId,
};

const client = new IotCloudClient(clientConfig);

client.connect()
  .then(() => client.retrieveDeviceList())
  .then((d) => {
    let cnt = 0;
    const listener = (state) => {
      console.log(state);
      cnt += 1;
      if (cnt > 1) {
        client.unsubscribeDevicePresence(d[0].id);
        client.removeListener(d[0].id, listener);
      }
    };
    client.on(d[0].id, listener);
    client.subscribeDevicePresence(d[0].id);
    return client.getDeviceResources(d[0].id);
  })
  .then((device) => {
    const href = device[0].links[0].href;
    const listener = (resources) => {
      console.log(resources);
      client.unsubscribeDeviceResource(href);
      client.removeListener(href, listener);
    };
    client.on(href, listener);
    client.subscribeDeviceResource(href);
  });
